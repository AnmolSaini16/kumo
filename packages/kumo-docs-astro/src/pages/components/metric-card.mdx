## Installation

### Barrel

```tsx
import { MetricCard, MetricCardGroup } from "@cloudflare/kumo";
```

### Granular

```tsx
import { MetricCard } from "@cloudflare/kumo/components/metric-card";
import { MetricCardGroup } from "@cloudflare/kumo/components/metric-card-group";
```

## Usage

```tsx
import { MetricCard, MetricCardGroup } from "@cloudflare/kumo";

export default function Example() {
  return (
    <MetricCardGroup title="Overview">
      <MetricCard label="Requests" value="1.2M" />
      <MetricCard label="Error Rate" value="0.3" unit="%" />
    </MetricCardGroup>
  );
}
```

## API Reference

### MetricCard

<p>
  Compact card displaying a single metric value with optional trend, sparkline,
  and loading/error states.
</p>
<PropsTable component="MetricCard" />

<p>The <a href="#trend-options"><code>trend</code></a> and <a href="#sparkline-options"><code>sparkline</code></a> props accept configuration objects described below.</p>

### Trend Options

<p>Options for the <code>trend</code> prop.</p>
<PropsTable props={{
  direction: { type: '"up" | "down" | "neutral"', description: 'Arrow direction: up (↑), down (↓), or neutral (→).', required: true },
  label: { type: "string", description: 'Pre-formatted trend label, e.g. "13.8%", "2.3pp", "5.1ms".', required: true },
  lessIsBetter: { type: "boolean", default: "false", description: "Inverts color semantics: down becomes green (good), up becomes red (bad). Use for metrics like latency or error count." },
  isNeutral: { type: "boolean", default: "false", description: "Forces neutral (gray) styling regardless of direction. Use for metrics where change has no good/bad meaning." },
}} />

### Sparkline Options

<p>Options for the <code>sparkline</code> prop.</p>
<PropsTable props={{
  data: { type: "number[]", description: "Array of numeric data points to plot.", required: true },
  theme: { type: '"success" | "danger" | "neutral"', default: "Brand color", description: 'Preset color theme for the sparkline. For custom colors beyond these presets, use the <code>color</code> prop instead.' },
  color: { type: "string", description: 'Custom color that overrides <code>theme</code>. Accepts any CSS color value including kumo tokens, e.g. <code>"var(--color-kumo-brand)"</code> or <code>"#E87B35"</code>.' },
  yMin: { type: "number", default: "Auto from data", description: "Override the auto-scaled minimum y value." },
  yMax: { type: "number", default: "Auto from data", description: "Override the auto-scaled maximum y value." },
}} />
